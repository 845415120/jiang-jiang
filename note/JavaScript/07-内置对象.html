1.内置对象 #

  • Javascript中的对象分为3种：自定义对象、内置对象、浏览器对象

2.4  Math.random 随机数 #

Math.random 方法返回一个位于区间 [0, 1) 的伪随机浮点数。

javascript
Math.random(); // 0.41713485547506424
1

获取闭区间 [a, b] 之间的整数：

javascript
let ran = parseInt(Math.random() * (b - a + 1)) + a;
1

获取前闭后开区间 [a, b) 之间的整数：

javascript
let ran = parseInt(Math.random() * (b - a)) + a;
1

实例：定义自己的取整函数

javascript
// [a, b)
function getRandom1(a, b) {
    return parseInt(Math.random() * (b -a)) + a;
}

// [a, b]
function getRandom2(a, b) {
    return parseInt(Math.random() * (b - a + 1)) + a;
}

4.7 查找某个字符出现的位置和次数 #

javascript
// 查找某个字符出现的位置和次数
function findLocation(str, char) {
    let res = {
        loc: [],
        count: 0
    };
    for (let i = 0; i < str.length; i++) {
        if (str[i] === char) {
            res.loc.push(i);
            res.count++;
        }
    }
    return res;
}

let ans = findLocation("abcaaadef", "a");
console.log(ans.count, ans.loc); // 4 [ 0, 3, 4, 5 ]

4.8 统计出现次数最多的字符 #

javascript
function findMost(str) {
    let res = {};
    for (const i in str) {
        if (str[i] in res) {
            res[str[i]]++;
        } else {
            res[str[i]] = 1;
        }
    }
    let temp = -Infinity;
    let ans;
    for (const k in res) {
        if (res[k] > temp) {
            temp = res[k];
            ans = k;
        }
    }
    return [ans, temp];
};
console.log(findMost("abbcaaa")); // [ 'a', 4 ]
